71 void Processor::handleMessage(cMessage *msg)
72 {
73  if(strcmp("networkDiscovery",msg->par("type").stringValue()) == 0)
74  {
75  MacTableEntry *macTableEntry = new MacTableEntry();
76  macTableEntry->setIdentity(switchMemory->macEntryIndex++);
77  macTableEntry->setInterface(msg->par("interface").stringValue());
78  macTableEntry->setMacAddress(msg->par("macAddress").stringValue());
79  macTableEntry->setQuantumInterfaceId(" ");
80  macTableEntry->setQuantumMacAddress(" ");
81  macTableEntry->setType(1);

83 
84  ArpTableEntry *arpTableEntry = new ArpTableEntry();
85  arpTableEntry->setIdentity(switchMemory->arpEntryIndex++);
86  arpTableEntry->setProtocol("Internet");
87  arpTableEntry->setIpAddress(msg->par("ipAddress").stringValue());

89  arpTableEntry->setMacAddress(msg->par("macAddress").stringValue());
90  arpTableEntry->setType(0);
91  arpTableEntry->setInterface(msg->par("interface").stringValue());
92  switchMemory->addArpTableEntry(arpTableEntry);
93 
94  delete msg;
95  }
96  else if (strcmp("initQkd",msg->par("type").stringValue()) == 0)
97  {
98  SessionStateEntry *sessionStateEntry = new SessionStateEntry();
99  sessionStateEntry->setSessionId(switchMemory->sessionsStateEntryIndex++);
100  sessionStateEntry->setSrcMac(msg->par("srcInterfaceMacAddress").stringValue());
101  sessionStateEntry->setSrcQMac(switchMemory->getQuantumInterfaceMacAddressFromMacTable(sessionStateEntry->getSrcMac()).c_str());
102  sessionStateEntry->setSrcSubInterface(switchMemory->getQuantumInterfaceFromMacTable(sessionStateEntry->getSrcQMac()).c_str());
103  sessionStateEntry->setDesMac(switchMemory->getMacAddressOfInterfaceFromMacTable(switchMemory->getInterfaceIdFromMacTable(msg->par("desMAC").stringValue()).c_str()).c_str());
104  sessionStateEntry->setDesQMac(switchMemory->getQuantumMacAddressOfInterfaceFromMacTable(switchMemory->getInterfaceIdFromMacTable(msg->par("desMAC").stringValue()).c_str()).c_str());
105  sessionStateEntry->setDesSubInterface(switchMemory->getQuantumInterfaceFromMacTable(sessionStateEntry->getDesQMac()).c_str());
106  sessionStateEntry->setExpiry("0000");
107  sessionStateEntry->setStatus("0");
108  if(switchMemory->entryExist(sessionStateEntry->getSrcMac(), sessionStateEntry->getDesMac()) == false)
109  {
110  switchMemory->addSessionStateTableEntry(sessionStateEntry);
111  cPacket *qkdSession = new cPacket("qkdRequest");
112  qkdSession->addPar("type").setStringValue("qkdRequest");
113  qkdSession->addPar("srcMAC").setStringValue(sessionStateEntry->getSrcMac());
114  qkdSession->addPar("requestFrom").setStringValue(msg->par("srcMAC").stringValue());
115  qkdSession->addPar("desMAC").setStringValue(sessionStateEntry->getDesMac());
116  qkdSession->addPar("requestTo").setStringValue(msg->par("desMAC").stringValue());
117  qkdSession->addPar("interface").setStringValue(switchMemory->getInterfaceIdFromMacTable(sessionStateEntry->getDesMac()).c_str());
118  send(qkdSession,"publicInterfaceCommunication$o");
119  }
120  else
121  {
122  cPacket *qkdSession = new cPacket("clientBusy");
123  qkdSession->addPar("type").setStringValue("initQkdRespons");
124  qkdSession->addPar("desMAC").setStringValue(sessionStateEntry->getDesMac());
125  qkdSession->addPar("interface").setStringValue(switchMemory->getInterfaceIdFromMacTable(sessionStateEntry->getSrcMac()).c_str());
126  send(qkdSession,"publicInterfaceCommunication$o");
127  }
128  delete msg;
129  }
130  else if(strcmp("QKD-RESPONSE",msg->par("type").stringValue()) == 0)
131  {
132  cPacket *qkdAck = new cPacket("qkdAck");
133  qkdAck->addPar("type").setStringValue("QKD-ACK");
134  qkdAck->addPar("srcMAC").setStringValue(msg->par("srcMAC").stringValue());
135  qkdAck->addPar("desMAC").setStringValue(msg->par("desMAC").stringValue());
136  qkdAck->addPar("interface").setStringValue(switchMemory->getInterfaceIdFromMacTable(msg->par("desMAC").stringValue()).c_str());
137 
138  cModule *qSRCInterface = this->getParentModule()->getSubmodule(switchMemory->getQuantumInterfaceFromMacTableByPublicMac(msg->par("desMAC").stringValue()).c_str());
139  cModule *qDESInterface = this->getParentModule()->getSubmodule(switchMemory->getQuantumInterfaceFromMacTableByPublicMac(msg->par("srcMAC").stringValue()).c_str());
140  qSRCInterface->par("exitInterface").setStringValue(switchMemory->getExitInterfaceFromBindingTable(qSRCInterface->getName(), qDESInterface->getName()));
141 
142  send(qkdAck,"publicInterfaceCommunication$o");
143 
144  delete msg;
145  }
146  else if (strcmp("quantumData",msg->par("type").stringValue()) == 0)
147  {
148  cPacket *pkt = new cPacket("qkdStats");
149  pkt->addPar("type").setStringValue("qkdStates");
150  pkt->addPar("interface").setStringValue(switchMemory->getInterface(msg->par("desMAC").stringValue()).c_str());
151  pkt->addPar("srcMAC").setStringValue(msg->par("srcMAC").stringValue());
152  pkt->addPar("desMAC").setStringValue(msg->par("desMAC").stringValue());
153  pkt->addPar("filterUsage").setStringValue(msg->par("filterUsage").stringValue());
154  send(pkt,"publicInterfaceCommunication$o");
155 
156  delete msg;
157  }
158  else
159  {
160  delete msg;
161  }
